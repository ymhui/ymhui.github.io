smartupload上传文件使用

注意点

  • smartupload –> getParameter 获取参数中文乱码，将eclipse编码改为utf-8即可

上传代码

代码模板

package com.onlineshop.product;

import java.io.IOException;
import java.io.PrintWriter;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

import com.jspsmart.upload.File;
import com.jspsmart.upload.Files;
import com.jspsmart.upload.Request;
import com.jspsmart.upload.SmartUpload;
import com.jspsmart.upload.SmartUploadException;

import Bean.Product;
import Dao.ProductDao;

/**
 * Servlet implementation class DoproductAdd
 */
@WebServlet("/manage/admin_doproductadd")
public class DoproductAdd extends HttpServlet {
    private static final long serialVersionUID = 1L;

    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        request.setCharacterEncoding("utf-8"); //1
        response.setContentType("text/html;charset=utf-8");  //2
        // 创建SmartUpload对象
        SmartUpload su = new SmartUpload();
        // 初始化
        su.initialize(this.getServletConfig(), request, response);
        // 上传过程
        try {
            su.upload();
        } catch (SmartUploadException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
        // 获取上传对象
        Files files = su.getFiles();
        File file = files.getFile(0);
        // 获取上传文件名
        String filename = file.getFileName();
        String path=this.getServletContext().getRealPath("")+"\\images\\product";
        try {
            //这里应该存绝对路径，不然每次重启tomcat会删除图片
            su.save(path);
            //su.save("D:\\insapp\\apache-tomcat-7.0.105\\webapps\\Online_Shop\\images\\product");
        } catch (SmartUploadException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
//        System.out.println(filename);
        // 有上传文件只能通过这样获取request
        Request req1 = su.getRequest();

        String pname =new String(req1.getParameter("productName").getBytes(),"UTF-8");
        System.out.println(pname);
        String pprice = req1.getParameter("productPrice");
        String pstock = req1.getParameter("productStock");
        String parid = req1.getParameter("parid");
        String pdesc = new String(req1.getParameter("productDesc").getBytes(),"UTF-8");
        System.out.println(pdesc);
        Product product = new Product(0, pname, pdesc, Double.parseDouble(pprice), Integer.parseInt(pstock),
                Integer.parseInt((parid.split("-")[0])), Integer.parseInt((parid.split("-")[1])), filename);
        int res = new ProductDao().insert(product);
        PrintWriter out = response.getWriter();
        if (res > 0) {
            out.print("<script>");
            out.print("alert('图书插入成功！');");
            out.print("location.href='admin_doproductselect';");
            out.print("</script>");
            return;
        }
        out.print("<script>");
        out.print("alert('图书插入失败！');");
        out.print("location.href='admin_toproductadd';");
        out.print("</script>");
    }

    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        // TODO Auto-generated method stub
        doGet(request, response);
    }
}

下载代码

package com.onlineshop.product;

import java.io.IOException;


import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

import com.jspsmart.upload.SmartUpload;
import com.jspsmart.upload.SmartUploadException;

/**
 * Servlet implementation class DoDownLoad
 */
@WebServlet("/manage/dodownload")
public class DoDownLoad extends HttpServlet {
    private static final long serialVersionUID = 1L;

    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
         //1、创建SmartUpload对象
        SmartUpload smartUpload = new SmartUpload();

        //2、初始化创建的SmartUpload对象
        smartUpload.initialize(getServletConfig(), request, response);

        //3、文件下载（同时需要得到文件的下载路径。通过 下载的jsp页面的a连接的URL？后面的参数获得对应的文件名）
        String filename = request.getParameter("filename");         //获得要下载的文件名
        String fileSavePath = request.getServletContext().getRealPath("")+"\\images\\product";    //因为文件是保存在服务器的upload文件夹中的，所以要获得文件夹的路径
        String lastDownPath = fileSavePath + "/" + filename;    //得到文件下载的最终路径
        try {
            smartUpload.downloadFile(lastDownPath);             //进行下载
        } catch (SmartUploadException e) {
            e.printStackTrace();
        }
    }

    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        // TODO Auto-generated method stub
        doGet(request, response);
    }

}
